  {\normalfont\fontsize{14}{16.8}\bfseries\centering}
  {\chaptername\ \thechapter}
  {6pt}

  {0pt}
  {\MakeUppercase}

  {\normalfont\fontsize{13}{15.6}\bfseries}
  {\thesection.}{0.5em}{}

% Pod-podrozdziały: 12pt, pogrubione
\titleformat{\subsection}
  {\normalfont\fontsize{12}{14.4}\bfseries}
